\def\pizahl{3.14159}
\pgfmathparse{8/(\pizahl*\pizahl)}
\xdef\a{\pgfmathresult}
\pgfmathparse{180/\pizahl}
\xdef\xs{\pgfmathresult}

\def\eins{
\draw[color=red] plot[domain=0:{2*\pizahl},samples=100]
	({\x},{\a*(sin(\xs*\x))});
}

\def\zwei{
\draw[color=red] plot[domain=0:{2*\pizahl},samples=100]
	({\x},{\a*(sin(\xs*\x)-sin(3*\xs*\x)/9)});
}

\def\drei{
\draw[color=red] plot[domain=0:{2*\pizahl},samples=100]
	({\x},{\a*(sin(\xs*\x)-sin(3*\xs*\x)/9+sin(5*\xs*\x)/25)});
}

\def\vier{
\draw[color=red] plot[domain=0:{2*\pizahl},samples=100]
	({\x},{\a*(sin(\xs*\x) -sin(3*\xs*\x)/9 +sin(5*\xs*\x)/25 -sin(7*\xs*\x)/49)});
}

\def\fuenf{
\draw[color=red] plot[domain=0:{2*\pizahl},samples=100]
	({\x},{\a*(sin(\xs*\x) -sin(3*\xs*\x)/9 +sin(5*\xs*\x)/25 -sin(7*\xs*\x)/49 +sin(9*\xs*\x)/81)});
}

\def\sechs{
\draw[color=red] plot[domain=0:{2*\pizahl},samples=100]
	({\x},{\a*(sin(\xs*\x) -sin(3*\xs*\x)/9 +sin(5*\xs*\x)/25 -sin(7*\xs*\x)/49 +sin(9*\xs*\x)/81 -sin(11*\xs*\x)/121)});
}

\def\sieben{
\draw[color=red] plot[domain=0:{2*\pizahl},samples=100]
	({\x},{\a*(sin(\xs*\x) -sin(3*\xs*\x)/9 +sin(5*\xs*\x)/25 -sin(7*\xs*\x)/49 +sin(9*\xs*\x)/81 -sin(11*\xs*\x)/121+sin(13*\xs*\x)/169)});
}

\def\acht{
\draw[color=red] plot[domain=0:{2*\pizahl},samples=100]
	({\x},{\a*(sin(\xs*\x) -sin(3*\xs*\x)/9 +sin(5*\xs*\x)/25 -sin(7*\xs*\x)/49 +sin(9*\xs*\x)/81 -sin(11*\xs*\x)/121+sin(13*\xs*\x)/169-sin(15*\xs*\x)/225)});
}

\def\raster#1#2{
\draw[->] (-0.1,0)--(6.5,0) coordinate[label=$t$];
\draw[->] (0,-1.05)--(0,1.15) coordinate[label={right:$y$}];

\draw ({1*\pizahl/2},-0.02)--({1*\pizahl/2},0.02);
\node at ({1*\pizahl/2},0) [below] {$\displaystyle\frac{\mathstrut\pi}2$};
\draw ({2*\pizahl/2},-0.02)--({2*\pizahl/2},0.02);
\node at ({2*\pizahl/2},0) [below] {$\displaystyle\mathstrut\pi$};
\draw ({3*\pizahl/2},-0.02)--({3*\pizahl/2},0.02);
\node at ({3*\pizahl/2},0) [below] {$\displaystyle\frac{3\mathstrut\pi}2$};
\draw ({4*\pizahl/2},-0.02)--({4*\pizahl/2},0.02);
\node at ({4*\pizahl/2},0) [below] {$\displaystyle2\mathstrut\pi$};

\draw (-0.02,1)--(0.02,1);
\node at (0,1) [left] {$1$};
\draw (-0.02,-1)--(0.02,-1);
\node at (0,-1) [left] {$-1$};

\draw[color=blue!40,line width=1.4pt] (0,0)--({\pizahl/2},1)--({\pizahl},0)--({1.5*\pizahl},-1)--({2*\pizahl},0);

\csname#2\endcsname

\pgfplotsset{
/pgf/number format/textnumber/.style={fixed zerofill,precision=0}
}

\node at (5.6,1.0) [below right] {$n= \pgfmathprintnumber[textnumber]{#1}$};

}

\raster{1}{eins}

\begin{scope}[yshift = -2.3cm]
\raster{2}{zwei}
\end{scope}

\begin{scope}[yshift = -4.6cm]
\raster{3}{drei}
\end{scope}

\begin{scope}[yshift = -6.9cm]
\raster{8}{acht}
\end{scope}
