\section{Analytische Lösung}

\subsection{Schrittweises Lösen}
\begin{frame}
	\frametitle{Schrittweises Lösen einer DDE}
	\begin{itemize}
		\item[] Beispiel
			\begin{equation}
			\dot{y}(t)=ky(t-\tau)=-y(t-1)
			\end{equation} 
			\pause
		\item[] Anfangsbedingung
			\begin{equation}
			y(t)=1 \textrm{ if }	-1\le t<0
			\end{equation}
			\pause
		\item[] Lösung im Bereich von 0 bis 1 
		 	\pause
		 	\begin{equation}
		 	\dot{y}=-1 \textrm{   daraus folgt  } y=1-t
		 	\end{equation} 	
	\end{itemize}
\end{frame}

\subsection{Charakteristische Gleichung}
\begin{frame}
\frametitle{Charakteristische Gleichung}
\begin{itemize}
	\item[] Beispiel
	\begin{equation}
	\dot{y}(t)=ky(t-\tau)
	\end{equation} 
	\item[] Ansatz
	\begin{equation}
	y(t) = ce^{\lambda t}
	\end{equation}
	\pause
	\item[] Einsetzen
	\begin{equation}
	\lambda ce^{\lambda t} = kce^{\lambda (t-\tau )}
	\end{equation} 	
	\pause
	\item[] Vereinfachen
	\begin{equation}
	\lambda  - ke^{-\lambda}= 0
	\end{equation}
\end{itemize}
\end{frame}
